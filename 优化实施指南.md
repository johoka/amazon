```javascript
// 创建 utils/storage.js
class StorageUtil {
  static async get(key, defaultValue = null) {
    return new Promise((resolve, reject) => {
      chrome.storage.local.get(key, (result) => {
        if (chrome.runtime.lastError) {
          reject(chrome.runtime.lastError);
        } else {
          resolve(result[key] ?? defaultValue);
        }
      });
    });
  }

  static async set(key, value) {
    return new Promise((resolve, reject) => {
      chrome.storage.local.set({ [key]: value }, () => {
        if (chrome.runtime.lastError) {
          reject(chrome.runtime.lastError);
        } else {
          resolve();
        }
      });
    });
  }
}

// 使用方式
```

```javascript
// 创建 utils/tabManager.js
class TabManager {
  constructor() {
    this.tabListeners = new Map();
    this.pendingTabs = new Map();
  }

  async createAndWaitForComplete(url, orderId, senderTabId) {
    return new Promise((resolve, reject) => {
      chrome.tabs.create({ url, active: false }, (newTab) => {
        const listener = (tabId, changeInfo, updatedTab) => {
          if (tabId === newTab.id && changeInfo.status === "complete") {
            // 立即移除监听器
            chrome.tabs.onUpdated.removeListener(listener);
            this.tabListeners.delete(newTab.id);

            // 发送消息
            chrome.tabs.sendMessage(newTab.id, {
              startInvite: true,
              orderId: orderId,
              parentTabId: senderTabId,
            });

            resolve(newTab);
          }
        };

        // 保存监听器引用用于清理
        chrome.tabs.onUpdated.addListener(listener);
        this.tabListeners.set(newTab.id, listener);

        // 超时清理
        const timeoutId = setTimeout(() => {
          chrome.tabs.onUpdated.removeListener(listener);
          this.tabListeners.delete(newTab.id);
          reject(new Error(`Tab ${newTab.id} creation timeout`));
        }, 40000);

        this.pendingTabs.set(newTab.id, timeoutId);
      });
    });
  }

  removeListener(tabId) {
    const listener = this.tabListeners.get(tabId);
    if (listener) {
      chrome.tabs.onUpdated.removeListener(listener);
      this.tabListeners.delete(tabId);
    }

    const timeoutId = this.pendingTabs.get(tabId);
    if (timeoutId) {
      clearTimeout(timeoutId);
      this.pendingTabs.delete(tabId);
    }
  }
}

const tabManager = new TabManager();
```

```javascript
class PageState {

    this.reset();
  }

    this.orderMap = new Map(); // orderId -> DOM element
    this.invitedOrderIds = new Set(); // 本页已邀请订单

    this.orderMap.set(orderId, element);

    this.invitedOrderIds.add(orderId);
  }

  markAsExcluded(orderId) {
    this.excludedOrderIds.add(orderId);
  }

    if (
      !this.invitedOrderIds.has(orderId) &&
      !this.excludedOrderIds.has(orderId)
    ) {
      this.pendingOrderIds.push(orderId);
    }
  }

  getNextOrder() {
    return this.pendingOrderIds.shift();
  }

  incrementTabCount() {
    this.activeTabCount++;
  }

  decrementTabCount() {
    this.activeTabCount = Math.max(0, this.activeTabCount - 1);
  }

  canOpenNewTab(maxTabs) {
    return this.activeTabCount < maxTabs;
  }

  getStats() {
    return {
      totalOrders: this.orderMap.size,
      invitedCount: this.invitedOrderIds.size,
      excludedCount: this.excludedOrderIds.size,
      pendingCount: this.pendingOrderIds.length,
      activeTabCount: this.activeTabCount,
    };
  }

  // 用于调试
  toJSON() {
    return {
      orderMapSize: this.orderMap.size,
      invitedOrderIds: Array.from(this.invitedOrderIds),
      excludedOrderIds: Array.from(this.excludedOrderIds),

}
```

```javascript
// 创建 utils/elementWaiter.js
class ElementWaiter {
  async waitForElement(selector, timeout = 30000) {
    const startTime = Date.now();

    while (Date.now() - startTime < timeout) {
      const element = document.querySelector(selector);
      if (element) return element;
      await sleep(500);
    }

    throw new Error(`Element not found: ${selector}`);
  }

  async waitForAnyElement(selectors, timeout = 30000) {
    const startTime = Date.now();

    while (Date.now() - startTime < timeout) {
      for (const selector of selectors) {
        const element = document.querySelector(selector);
        if (element) return { element, selector };
      }
      await sleep(500);
    }

    throw new Error(`No element found for selectors: ${selectors.join(", ")}`);
  }
}

const elementWaiter = new ElementWaiter();
```

### logger.js - 统一日志系统

```javascript
// src/common/logger.js

const LOG_LEVEL = {
  DEBUG: 0,
  INFO: 1,
  WARN: 2,
  ERROR: 3,
};

class Logger {
  constructor(level = LOG_LEVEL.INFO) {
    this.level = level;
    this.logs = [];
    this.maxLogs = 1000;
  }

  log(level, message, data = null) {
    if (level < this.level) return;

    const timestamp = new Date().toISOString();
    const levelName = Object.keys(LOG_LEVEL).find(
      (key) => LOG_LEVEL[key] === level,
    );

    const logEntry = {
      timestamp,
      level: levelName,
      message,
      data,
    };

    // 输出到控制台
    const style = this.getConsoleStyle(level);
    const output = `[${timestamp}] ${levelName}: ${message}`;
    console.log(`%c${output}`, style, data);

    // 保存到内存
    this.logs.push(logEntry);
    if (this.logs.length > this.maxLogs) {
      this.logs.shift();
    }

    // 关键日志保存到storage
    if (level >= LOG_LEVEL.WARN) {
      this.saveToStorage(logEntry);
    }
  }

  debug(message, data) {
    this.log(LOG_LEVEL.DEBUG, message, data);
  }

  info(message, data) {
    this.log(LOG_LEVEL.INFO, message, data);
  }

  warn(message, data) {
    this.log(LOG_LEVEL.WARN, message, data);
  }

  error(message, error) {
    this.log(LOG_LEVEL.ERROR, message, {
      name: error?.name,
      message: error?.message,
      stack: error?.stack,
    });
  }

  getConsoleStyle(level) {
    const styles = {
      [LOG_LEVEL.DEBUG]: "color: gray;",
      [LOG_LEVEL.INFO]: "color: blue;",
      [LOG_LEVEL.WARN]: "color: orange;",
      [LOG_LEVEL.ERROR]: "color: red; font-weight: bold;",
    };
    return styles[level] || "";
  }

  saveToStorage(logEntry) {
    // 保存最近的错误日志到storage用于调试
    chrome.storage.local.get("logs", (result) => {
      const logs = result.logs || [];
      logs.push(logEntry);
      if (logs.length > 100) {
        logs.shift();
      }
      chrome.storage.local.set({ logs });
    });
  }

  exportLogs() {
    return JSON.stringify(this.logs, null, 2);
  }

  clearLogs() {
    this.logs = [];
    chrome.storage.local.remove("logs");
  }
}

export const logger = new Logger(LOG_LEVEL.INFO);
```

### retry.js - 重试机制

```javascript
// src/utils/retry.js

import { CONFIG } from "../common/constants.js";

class RetryStrategy {
  constructor(maxAttempts = 3, baseDelay = 1000, maxDelay = 30000) {
    this.maxAttempts = maxAttempts;
    this.baseDelay = baseDelay;
    this.maxDelay = maxDelay;
  }

  async execute(fn, context = {}) {
    let lastError;

    for (let attempt = 0; attempt < this.maxAttempts; attempt++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;

        const delay = Math.min(
          this.baseDelay * Math.pow(2, attempt),
          this.maxDelay,
        );

        if (attempt < this.maxAttempts - 1) {
          context.logger?.warn(
            `Retry ${attempt + 1}/${this.maxAttempts - 1} in ${delay}ms`,
            { error: error.message },
          );
          await sleep(delay);
        }
      }
    }

    throw lastError;
  }
}

// 预设策略

  SLOW: new RetryStrategy(5, 2000), // 缓慢重试
```

```javascript
// src/background/messageRouter.js

import { logger } from "../common/logger.js";

class MessageRouter {
  constructor() {
    this.handlers = {};
    this.setupListeners();
  }

  register(action, handler) {
    if (this.handlers[action]) {
      logger.warn(`Handler already registered for action: ${action}`);
    }
    this.handlers[action] = handler;
  }

  setupListeners() {
    chrome.runtime.onMessage.addListener((request, sender, sendResponse) => {
      const { action } = request;

      if (!action) {
        logger.warn("Message received without action", request);
        return;
      }

      const handler = this.handlers[action];
      if (!handler) {
        logger.warn(`No handler registered for action: ${action}`);
        return;
      }

      Promise.resolve(handler(request, sender))
        .then((response) => {
          sendResponse({ success: true, data: response });
        })
        .catch((error) => {
          logger.error(`Handler error for action: ${action}`, error);
          sendResponse({
            success: false,
            error: error.message,
            action: action,
          });
        });

      // 返回 true 表示异步响应
      return true;
    });
  }
}

    request.url,

  await storageUtil.recordInvitedOrder(request.orderId);

});
```
